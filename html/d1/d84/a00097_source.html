149 # ifdef MACE_OPENCV
150  void load(cv::Mat mat) {
151  //cv::flip(mat, mat, 0);
152 
153  Texture::Format colorFormat = Texture::Format::BGR;
154  if (mat.channels() == 1) {
155  colorFormat = Texture::Format::LUMINANCE;
156  } else if (mat.channels() == 2) {
157  colorFormat = Texture::Format::LUMINANCE_ALPHA;
158  } else if (mat.channels() == 4) {
159  colorFormat = Texture::Format::BGRA;
160  }
161 
162  Texture::Type type = Texture::Type::UNSIGNED_BYTE;
163  if (mat.depth() == CV_8S) {
164  type = Texture::Type::BYTE;
165  } else if (mat.depth() == CV_16U) {
166  type = Texture::Type::UNSIGNED_SHORT;
167  } else if (mat.depth() == CV_16S) {
168  type = Texture::Type::SHORT;
169  } else if (mat.depth() == CV_32S) {
170  type = Texture::Type::INT;
171  } else if (mat.depth() == CV_32F) {
172  type = Texture::Type::FLOAT;
173  } else if (mat.depth() == CV_64F) {
174  MACE__THROW(BadFormat, "Unsupported cv::Mat depth: CV_64F");
175  }
176 
177  resetPixelStorage();
178  setPackStorageHint(Texture::PixelStorage::ALIGNMENT, (mat.step & 3) ? 1 : 4);
179  setPackStorageHint(Texture::PixelStorage::ROW_LENGTH, static_cast<int>(mat.step / mat.elemSize()));
180 
181  setData(mat.ptr(), mat.cols, mat.rows, type, colorFormat, Texture::InternalFormat::RGBA);
182  }

190  cv::Mat img(height, width, CV_8UC3);

194  setPackStorageHint(Texture::PixelStorage::ROW_LENGTH, static_cast<int>(img.step / img.elemSize()));
195 
196  bind();
197 
198  getImage(Texture::Format::BGR, Texture::Type::UNSIGNED_BYTE, img.data);
199 
200  cv::flip(img, img, 0);
201 
202  return img;
203  }
