125  //this needs to be defined in the header file to prevent linker conflicts, because Entity.cpp does not have opencv included ever.
126 # ifdef MACE_OPENCV
127  void load(cv::Mat mat) {
128  //cv::flip(mat, mat, 0);
129 
130  Enum colorFormat = GL_BGR;
131  if (mat.channels() == 1) {
132  colorFormat = GL_LUMINANCE;
133  } else if (mat.channels() == 2) {
134  colorFormat = GL_LUMINANCE_ALPHA;
135  } else if (mat.channels() == 4) {
136  colorFormat = GL_BGRA;
137  }
138 
139  Enum type = GL_UNSIGNED_BYTE;
140  if (mat.depth() == CV_8S) {
141  type = GL_BYTE;
142  } else if (mat.depth() == CV_16U) {
143  type = GL_UNSIGNED_SHORT;
144  } else if (mat.depth() == CV_16S) {
145  type = GL_SHORT;
146  } else if (mat.depth() == CV_32S) {
147  type = GL_INT;
148  } else if (mat.depth() == CV_32F) {
149  type = GL_FLOAT;
150  } else if (mat.depth() == CV_64F) {
151  throw BadImageError("Unsupported cv::Mat depth: CV_64F");
152  }
153 
154  resetPixelStorage();
155  setPixelStorage(GL_PACK_ALIGNMENT, (mat.step & 3) ? 1 : 4);
156  setPixelStorage(GL_PACK_ROW_LENGTH, static_cast<int>(mat.step / mat.elemSize()));
157 
158  setData(mat.ptr(), mat.cols, mat.rows, type, colorFormat, GL_RGBA);
159 
160  ogl::checkGLError(__LINE__, __FILE__, "Error loading texture from OpenCV Mat");
161  }
162 
163  ColorAttachment(cv::Mat mat) : ColorAttachment() {
164  init();
165  load(mat);
166  }

168  cv::Mat toMat(const Size width, const Size height) {
169  cv::Mat img(height, width, CV_8UC3);
170 
171  resetPixelStorage();
172  setPixelStorage(GL_PACK_ALIGNMENT, (img.step & 3) ? 1 : 4);
173  setPixelStorage(GL_PACK_ROW_LENGTH, static_cast<int>(img.step / img.elemSize()));
174 
175  bind();
176 
177  glReadPixels(0, 0, img.cols, img.rows, GL_BGR, GL_UNSIGNED_BYTE, img.data);
178 
179  cv::flip(img, img, 0);
180 
181  return img;
182  }
