127  //this needs to be defined in the header file to prevent linker conflicts, because Entity.cpp does not have opencv included ever.
128 # ifdef MACE_OPENCV
129  void load(cv::Mat mat) {
130  //cv::flip(mat, mat, 0);
131 
132  Enum colorFormat = GL_BGR;
133  if (mat.channels() == 1) {
134  colorFormat = GL_LUMINANCE;
135  } else if (mat.channels() == 2) {
136  colorFormat = GL_LUMINANCE_ALPHA;
137  } else if (mat.channels() == 4) {
138  colorFormat = GL_BGRA;
139  }
140 
141  Enum type = GL_UNSIGNED_BYTE;
142  if (mat.depth() == CV_8S) {
143  type = GL_BYTE;
144  } else if (mat.depth() == CV_16U) {
145  type = GL_UNSIGNED_SHORT;
146  } else if (mat.depth() == CV_16S) {
147  type = GL_SHORT;
148  } else if (mat.depth() == CV_32S) {
149  type = GL_INT;
150  } else if (mat.depth() == CV_32F) {
151  type = GL_FLOAT;
152  } else if (mat.depth() == CV_64F) {
153  throw BadImageError("Unsupported cv::Mat depth: CV_64F");
154  }
155 
156  resetPixelStorage();
157  setPixelStorage(GL_PACK_ALIGNMENT, (mat.step & 3) ? 1 : 4);
158  setPixelStorage(GL_PACK_ROW_LENGTH, static_cast<int>(mat.step / mat.elemSize()));
159 
160  setData(mat.ptr(), mat.cols, mat.rows, type, colorFormat, GL_RGBA);
161 
162  ogl::checkGLError(__LINE__, __FILE__, "Error loading texture from OpenCV Mat");
163  }

165  ColorAttachment(cv::Mat mat) : ColorAttachment() {
166  init();
167  load(mat);
168  }

170  cv::Mat toMat(const Size width, const Size height) {
171  cv::Mat img(height, width, CV_8UC3);
172 
173  resetPixelStorage();
174  setPixelStorage(GL_PACK_ALIGNMENT, (img.step & 3) ? 1 : 4);
175  setPixelStorage(GL_PACK_ROW_LENGTH, static_cast<int>(img.step / img.elemSize()));
176 
177  bind();
178 
179  glReadPixels(0, 0, img.cols, img.rows, GL_BGR, GL_UNSIGNED_BYTE, img.data);
180 
181  cv::flip(img, img, 0);
182 
183  return img;
184  }
