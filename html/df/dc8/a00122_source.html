534  template<typename T, Size N>
535  T det(const Matrix<T, N, N>& matrix) {
536  MACE_STATIC_ASSERT(N >= 2, "In order to retrieve the determinate of a Matrix, its size must be bigger than 1");
537  T sum = T();
538  Index counter = 0;
539  for( Index i = 0; i < N; ++i ) {
540  MACE_CONSTEXPR Size newMatrixSize = N - 1;
541  Matrix<T, newMatrixSize, newMatrixSize> m = Matrix<T, newMatrixSize, newMatrixSize>();
542  Index counterX = 0;
543  Index counterY = 0;
544  for( Index x = 0; x < N; ++x ) {
545  for( Index y = 1; y < N; ++y ) {
546  if( x != i ) {
547  m[counterX++][counterY] = matrix[x][y];
548  }
549  counterY++;
550 
551  }
552  counterY = 0;
553  }
554  T tempSum = ((counter % 2 == 0 ? -1 : 1) * matrix[i][0]);
555  if( newMatrixSize == 2 ) {
556  tempSum *= det<T>(matrix);
557  } else {
558  tempSum *= det<T, N>(matrix);
559  }
560  sum += tempSum;
561  ++counter;
562  }
563  return sum;
564  }
