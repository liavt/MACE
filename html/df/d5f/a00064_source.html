109  //this needs to be defined in the header file to prevent linker conflicts, because Entity.cpp does not have opencv included.
110 # ifdef MACE_OPENCV
111  void load(cv::Mat mat) {
112  //cv::flip(mat, mat, 0);
113 
114  Enum colorFormat = GL_BGR;
115  if (mat.channels() == 1) {
116  colorFormat = GL_LUMINANCE;
117  }
118  else if (mat.channels() == 2) {
119  colorFormat = GL_LUMINANCE_ALPHA;
120  }
121  else if (mat.channels() == 4) {
122  colorFormat = GL_BGRA;
123  }
124 
125  Enum type = GL_UNSIGNED_BYTE;
126  if (mat.depth() == CV_8S) {
127  type = GL_BYTE;
128  }
129  else if (mat.depth() == CV_16U) {
130  type = GL_UNSIGNED_SHORT;
131  }
132  else if (mat.depth() == CV_16S) {
133  type = GL_SHORT;
134  }
135  else if (mat.depth() == CV_32S) {
136  type = GL_INT;
137  }
138  else if (mat.depth() == CV_32F) {
139  type = GL_FLOAT;
140  }
141  else if (mat.depth() == CV_64F) {
142  throw BadImageError("Unsupported cv::Mat depth: CV_64F");
143  }
144 
145  resetPixelStorage();
146  setPixelStorage(GL_PACK_ALIGNMENT, (mat.step & 3) ? 1 : 4);
147  setPixelStorage(GL_PACK_ROW_LENGTH, mat.step / mat.elemSize());
148 
149  //setData(mat.data, mat.cols, mat.rows, GL_UNSIGNED_BYTE, GL_BGR, GL_RGBA);
150  setData(mat.ptr(), mat.cols, mat.rows, type, colorFormat, GL_RGBA);
151 
152  ogl::checkGLError(__LINE__, __FILE__);
153  }
