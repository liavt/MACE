127  //this needs to be defined in the header file to prevent linker conflicts, because Entity.cpp does not have opencv included ever.
128 # ifdef MACE_OPENCV
129  void load(cv::Mat mat) {
130  //cv::flip(mat, mat, 0);
131 
132  Enum colorFormat = GL_BGR;
133  if (mat.channels() == 1) {
134  colorFormat = GL_LUMINANCE;
135  }
136  else if (mat.channels() == 2) {
137  colorFormat = GL_LUMINANCE_ALPHA;
138  }
139  else if (mat.channels() == 4) {
140  colorFormat = GL_BGRA;
141  }
142 
143  Enum type = GL_UNSIGNED_BYTE;
144  if (mat.depth() == CV_8S) {
145  type = GL_BYTE;
146  }
147  else if (mat.depth() == CV_16U) {
148  type = GL_UNSIGNED_SHORT;
149  }
150  else if (mat.depth() == CV_16S) {
151  type = GL_SHORT;
152  }
153  else if (mat.depth() == CV_32S) {
154  type = GL_INT;
155  }
156  else if (mat.depth() == CV_32F) {
157  type = GL_FLOAT;
158  }
159  else if (mat.depth() == CV_64F) {
160  throw BadImageError("Unsupported cv::Mat depth: CV_64F");
161  }
162 
163  resetPixelStorage();
164  setPixelStorage(GL_PACK_ALIGNMENT, (mat.step & 3) ? 1 : 4);
165  setPixelStorage(GL_PACK_ROW_LENGTH, static_cast<int>(mat.step / mat.elemSize()));
166 
167  setData(mat.ptr(), mat.cols, mat.rows, type, colorFormat, GL_RGBA);
168 
169  ogl::checkGLError(__LINE__, __FILE__, "Error loading texture from OpenCV Mat");
170  }
