522  T det(const Matrix<T, N, N>& matrix) {
523  static_assert(N >= 2, "In order to retrieve the determinate of a Matrix, its size must be bigger than 1");
524  T sum = T();
525  Index counter = 0;
526  for( Index i = 0; i < N; i++ ) {
527  const Size newMatrixSize = N - 1;
528  Matrix<T, newMatrixSize, newMatrixSize> m = Matrix<T, newMatrixSize, newMatrixSize>();
529  Index counterX = 0;
530  Index counterY = 0;
531  for( Index x = 0; x < N; x++ ) {
532  for( Index y = 1; y < N; y++ ) {
533  if( x != i ) {
534  m[counterX][counterY] = matrix[x][y];
535  counterX++;
536  }
537  counterY++;
538 
539  }
540  counterY = 0;
541  }
542  T tempSum = ((counter % 2 == 0 ? -1 : 1) * matrix[i][0]);
543  if( N - 1 == 2 ) {
544  tempSum *= det<T>(matrix);

546  tempSum *= det<T, N>(matrix);
