499  class Ptr {
500  public:

502  Ptr( T* p ) : m_p( p ){
503  if( m_p )
504  m_p->addRef();
505  }
506  Ptr( Ptr const& other ) : m_p( other.m_p ){
507  if( m_p )
508  m_p->addRef();

512  m_p->release();
513  }

515  if( m_p )
516  m_p->release();
