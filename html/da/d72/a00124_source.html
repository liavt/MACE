567  T det(const Matrix<T, N, N>& matrix) {
568  static_assert(N >= 2, "In order to retrieve the determinate of a Matrix, its size must be bigger than 1");
569  T sum = T();
570  Index counter = 0;

572  constexpr Size newMatrixSize = N - 1;
573  Matrix<T, newMatrixSize, newMatrixSize> m = Matrix<T, newMatrixSize, newMatrixSize>();
574  Index counterX = 0;
575  Index counterY = 0;
576  for( Index x = 0; x < N; x++ ) {
577  for( Index y = 1; y < N; y++ ) {
578  if( x != i ) {
579  m[counterX][counterY] = matrix[x][y];
580  counterX++;
581  }
582  counterY++;
583 
584  }
585  counterY = 0;
586  }
587  T tempSum = ((counter % 2 == 0 ? -1 : 1) * matrix[i][0]);
588  if( newMatrixSize == 2 ) {
